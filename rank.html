// 示例：滚动触发排名动画，添加节流减少频繁触发 function throttle(fn, delay = 300) { let timer = null; return function(...args) { if (!timer) { timer = setTimeout(() => { fn.apply(this, args); timer = null; }, delay); } }; } // 滚动触发动画（节流优化） window.addEventListener("scroll", throttle(() => { const rankContainer = document.getElementById("rank-container"); const rect = rankContainer.getBoundingClientRect(); // 当容器进入视口时触发动画 if (rect.top < window.innerHeight && rect.bottom > 0) { rankContainer.classList.add("animate-active"); } }), 300);
